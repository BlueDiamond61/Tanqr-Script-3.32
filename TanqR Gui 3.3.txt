-- Improved Aimlock Button Logic
TANQRAIM.MouseButton1Down:Connect(function()
	for _,v in Players:GetPlayers() do
		if v ~= LocalPlayer and v.Character then
			local char = v.Character
			local parts = {"RightUpperLeg", "LeftUpperLeg", "HumanoidRootPart"}
			for i, partName in parts do
				local part = char:FindFirstChild(partName)
				if part and part:IsA("BasePart") then
					part.CanCollide = false
					part.Transparency = 1
					part.Size = Vector3.new(13,13,13)
				end
			end
			-- HeadHB is not standard, so check existence and type
			local headHB = char:FindFirstChild("HeadHB")
			if headHB and headHB:IsA("BasePart") then
				headHB.CanCollide = false
				headHB.Transparency = 1
				headHB.Size = Vector3.new(13,13,13)
			end
		end
	end
end)

-- Modern Draggable Frame Logic
local dragging = false
local dragInput, mousePos, framePos

Aimlock.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		mousePos = input.Position
		framePos = Aimlock.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

Aimlock.InputChanged:Connect(function(input)
	if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
		local delta = input.Position - mousePos
		Aimlock.Position = UDim2.new(
			framePos.X.Scale,
			framePos.X.Offset + delta.X,
			framePos.Y.Scale,
			framePos.Y.Offset + delta.Y
		)
	end
end)
